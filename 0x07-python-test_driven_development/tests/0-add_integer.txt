#!/usr/bin/python3
import unittest
from add_integer_module import add_integer

class TestAdd(unittest.TestCase):
    def test_add_positive_values(self):
        self.assertEqual(add_integer(2, 98), 100)
    
    def test_add_negative_values(self):
        self.assertEqual(add_integer(-2, 98), 96)

    def test_add_float_values(self):
        self.assertEqual(add_integer(2.5, 98), 100.5)

    def test_add_integer_default_value(self):
        result = add_integer(5)
        self.assertEqual(result, 103)  # 5 + 98 (default value for b)

    def test_add_integer_type_error_float_a(self):
        with self.assertRaises(TypeError):
            add_integer(5.5, 3)

    def test_add_integer_type_error_float_b(self):
        with self.assertRaises(TypeError):
            add_integer(5, 3.3)

    def test_add_integer_type_error_non_numeric_a(self):
        with self.assertRaises(TypeError):
            add_integer('5', 3)

    def test_add_integer_type_error_non_numeric_b(self):
        with self.assertRaises(TypeError):
            add_integer(5, '3')

if __name__ == '__main__':
    unittest.main()
